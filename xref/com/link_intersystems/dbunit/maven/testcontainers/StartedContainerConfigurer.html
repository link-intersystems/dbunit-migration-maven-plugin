18      private final GenericJdbcContainer genericJdbcContainer;
19      private final GenericContainerConfig containerConfig;
20      private final StringSearchInterpolator interpolator;

22      public StartedContainerConfigurer(GenericJdbcContainer genericJdbcContainer, GenericContainerConfig containerConfig) {
23          this.genericJdbcContainer = genericJdbcContainer;
24          this.containerConfig = containerConfig;
25  
26          DockerContainerConfig dockerContainerConfig = containerConfig.getDockerContainerConfig();
27          interpolator = new StringSearchInterpolator("{{", "}}");
28          interpolator.addValueSource(new DockerContainerConfigValueSource(dockerContainerConfig));
29          interpolator.addValueSource(new JdbcContainerValueSource(genericJdbcContainer));
30      }
31  
32      @Override
33      public void configure(GenericJdbcContainer container) {
34  
35          DataSourceConfig dataSourceConfig = containerConfig.getDataSourceConfig();
36  
37          String driverClassName = dataSourceConfig.getDriverClassName();
38          genericJdbcContainer.withDriverClassName(driverClassName);
39  
40          String jdbcUrl = getInterpolated(dataSourceConfig::getJdbcUrl);
41          logger.debug("JdbcUrl '{}' interpolated '{}'", dataSourceConfig.getJdbcUrl(), jdbcUrl);
42          genericJdbcContainer.withJdbcUrl(jdbcUrl);
43  
44          String username = getInterpolated(dataSourceConfig::getUsername);
45          logger.debug("Username '{}' interpolated '{}'", dataSourceConfig.getUsername(), username);
46          genericJdbcContainer.withUsername(username);
47  
48          String password = getInterpolated(dataSourceConfig::getPassword);
49          logger.debug("Password '{}' interpolated '{}'", dataSourceConfig.getPassword(), password);
50          genericJdbcContainer.withPassword(password);
51  
52          String testQueryString = getInterpolated(dataSourceConfig::getTestQueryString);
53          logger.debug("TestQueryString '{}' interpolated '{}'", dataSourceConfig.getTestQueryString(), testQueryString);
54          genericJdbcContainer.withTestQueryString(testQueryString);
55      }
56  
57      private String getInterpolated(Supplier<String> stringSupplier) {
58          String s = stringSupplier.get();
59          try {
60              return interpolator.interpolate(s);
61          } catch (InterpolationException e) {
62              throw new RuntimeException(e);
63          }
64      }
